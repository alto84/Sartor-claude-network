After onboarding, use the network like this:

from sartor_network_bootstrap import SartorNetworkClient

# Connect
client = SartorNetworkClient()
client.connect()

# Send a message
client.message_broadcast("Hello network!")

# Check for tasks
tasks = client.task_list()

# Share knowledge
client.knowledge_add("Important fact", tags=["category"])

# For sub-agents:
sub_prompt = client.get_sub_agent_prompt()
Task(prompt=sub_prompt + "\n\nYour task...", ...)
